CREATE DATABASE zimple;

USE zimple;

CREATE TABLE users(
id INT NOT NULL AUTO_INCREMENT, 
name VARCHAR(60),
lastname VARCHAR(60),
email VARCHAR(60),
password VARCHAR(60),

PRIMARY KEY(id));


CREATE TABLE products(
id INT NOT NULL AUTO_INCREMENT, 
name VARCHAR(60),
description VARCHAR(100),
image LONGBLOB,
price INT,
quantity INT,

PRIMARY KEY(id));

CREATE TABLE cart(
id INT NOT NULL AUTO_INCREMENT, 
idProduct INT,
idSession VARCHAR(60),
quantity INT,

PRIMARY KEY(id));

CREATE TABLE sales(
id INT NOT NULL AUTO_INCREMENT,
idProduct_Car INT, 
address VARCHAR(60),
phone VARCHAR(10),
date DATE,

PRIMARY KEY(id),
FOREIGN KEY(idProduct_Car) REFERENCES products(id) ON DELETE SET NULL ON UPDATE SET NULL
);
